## CSS Customization

### Custom CSS Location

Add custom CSS in one of these locations:

<Tabs>
  <Tab title="Plugin Settings">
    **Recommended for simple styles:**
    
    ```
    Generatify → Settings → Advanced → Custom CSS
    
    Paste your CSS here (no <style> tags needed)
    ```
    
    **Pros:**
    - Easy to manage
    - Survives plugin updates
    - No theme editing required
    
    **Cons:**
    - Limited to CSS only
  </Tab>
  
  <Tab title="Theme style.css">
    **For theme-specific styles:**
    
    ```css
    /* In your theme's style.css or custom CSS file */
    
    /* Generatify Custom Styles */
    .generatify-chat-widget {
        /* Your styles here */
    }
    ```
    
    **Pros:**
    - Integrated with theme
    - Can use theme variables
    
    **Cons:**
    - Lost on theme change
  </Tab>
  
  <Tab title="Child Theme">
    **Best practice for permanence:**
    
    ```css
    /* In child theme's style.css */
    
    /* Generatify Overrides */
    @import url('../generatify/assets/css/admin.min.css');
    
    .generatify-chat-widget {
        /* Your custom styles */
    }
    ```
    
    **Pros:**
    - Survives theme updates
    - Professional approach
    
    **Cons:**
    - Requires child theme setup
  </Tab>
</Tabs>

## WordPress Hooks & Filters

### Filters

Modify data before it's used:

<AccordionGroup>
  <Accordion title="generatify_ai_prompt" icon="message">
    **Modify AI prompts before sending:**
    
    ```php
    add_filter( 'generatify_ai_prompt', function( $prompt, $feature, $context ) {
        // Add custom instructions
        if ( $feature === 'chat' ) {
            $prompt .= "\nAlways be friendly and helpful.";
        }
        
        return $prompt;
    }, 10, 3 );
    ```
    
    **Parameters:**
    - `$prompt` (string): The prompt text
    - `$feature` (string): Feature name (chat, tags, categories, etc.)
    - `$context` (array): Additional context data
    
    **Return:** Modified prompt string
  </Accordion>
  
  <Accordion title="generatify_ai_response" icon="reply">
    **Filter AI responses:**
    
    ```php
    add_filter( 'generatify_ai_response', function( $response, $feature ) {
        // Remove certain phrases
        $response = str_replace( 'As an AI', '', $response );
        
        // Add custom suffix
        if ( $feature === 'chat' ) {
            $response .= "\n\nNeed more help? Contact us!";
        }
        
        return $response;
    }, 10, 2 );
    ```
    
    **Parameters:**
    - `$response` (string): AI response text
    - `$feature` (string): Feature name
    
    **Return:** Modified response string
  </Accordion>
  
  <Accordion title="generatify_generated_tags" icon="tags">
    **Modify auto-generated tags:**
    
    ```php
    add_filter( 'generatify_generated_tags', function( $tags, $post_id ) {
        // Always add a specific tag
        $tags[] = 'AI Generated';
        
        // Remove unwanted tags
        $tags = array_filter( $tags, function( $tag ) {
            return $tag !== 'unwanted';
        });
        
        // Limit to 5 tags
        return array_slice( $tags, 0, 5 );
    }, 10, 2 );
    ```
    
    **Parameters:**
    - `$tags` (array): Generated tag names
    - `$post_id` (int): Post ID
    
    **Return:** Modified tags array
  </Accordion>
  
  <Accordion title="generatify_generated_categories" icon="folder">
    **Modify auto-generated categories:**
    
    ```php
    add_filter( 'generatify_generated_categories', function( $categories, $post_id ) {
        // Map AI categories to your structure
        $mapping = [
            'Tech' => 'Technology',
            'Marketing' => 'Digital Marketing',
        ];
        
        foreach ( $categories as &$category ) {
            if ( isset( $mapping[$category] ) ) {
                $category = $mapping[$category];
            }
        }
        
        return $categories;
    }, 10, 2 );
    ```
  </Accordion>
  
  <Accordion title="generatify_meta_description" icon="file-lines">
    **Customize meta descriptions:**
    
    ```php
    add_filter( 'generatify_meta_description', function( $description, $post_id ) {
        // Add brand name
        $description .= ' | Your Brand Name';
        
        // Ensure length limit
        if ( strlen( $description ) > 160 ) {
            $description = substr( $description, 0, 157 ) . '...';
        }
        
        return $description;
    }, 10, 2 );
    ```
  </Accordion>
  
  <Accordion title="generatify_related_posts_query" icon="link">
    **Modify related posts query:**
    
    ```php
    add_filter( 'generatify_related_posts_query', function( $args, $post_id ) {
        // Only show posts from last 6 months
        $args['date_query'] = [
            'after' => '6 months ago',
        ];
        
        // Exclude specific categories
        $args['category__not_in'] = [5, 12];
        
        // Require featured image
        $args['meta_query'] = [
            [
                'key' => '_thumbnail_id',
                'compare' => 'EXISTS',
            ],
        ];
        
        return $args;
    }, 10, 2 );
    ```
  </Accordion>
  
  <Accordion title="generatify_toxicity_threshold" icon="shield-check">
    **Dynamic toxicity thresholds:**
    
    ```php
    add_filter( 'generatify_toxicity_threshold', function( $threshold, $category, $user_id ) {
        // Lower threshold for new users
        $user = get_userdata( $user_id );
        if ( get_comments( ['user_id' => $user_id, 'count' => true] ) < 3 ) {
            $threshold -= 20; // More strict for new users
        }
        
        // Adjust by time of day
        $hour = (int) current_time( 'H' );
        if ( $hour < 6 || $hour > 22 ) {
            $threshold -= 10; // More strict outside business hours
        }
        
        return $threshold;
    }, 10, 3 );
    ```
  </Accordion>
</AccordionGroup>

## Advanced Customization

### Custom AI Provider

Add your own AI provider:

```php
add_filter( 'generatify_ai_providers', function( $providers ) {
    $providers['custom'] = [
        'name' => 'Custom AI',
        'class' => 'My_Custom_AI_Provider',
        'settings' => [
            'api_key' => [
                'type' => 'text',
                'label' => 'API Key',
            ],
            'endpoint' => [
                'type' => 'text',
                'label' => 'API Endpoint',
            ],
        ],
    ];
    
    return $providers;
});

// Implement provider class
class My_Custom_AI_Provider {
    public function send_request( $prompt, $options = [] ) {
        // Your API integration code
        $response = wp_remote_post( $this->endpoint, [
            'headers' => [
                'Authorization' => 'Bearer ' . $this->api_key,
            ],
            'body' => json_encode([
                'prompt' => $prompt,
                'max_tokens' => $options['max_tokens'] ?? 1000,
            ]),
        ]);
        
        return json_decode( wp_remote_retrieve_body( $response ) );
    }
}
```

### Custom Knowledge Source

Add custom knowledge sources for chatbot:

```php
add_filter( 'generatify_knowledge_sources', function( $sources, $context ) {
    // Add external API data
    if ( $context['page_type'] === 'product' ) {
        $product_id = $context['product_id'];
        $product_data = fetch_external_product_data( $product_id );
        $sources[] = [
            'type' => 'external_api',
            'content' => $product_data,
        ];
    }
    
    // Add custom post type
    $custom_posts = get_posts([
        'post_type' => 'documentation',
        'posts_per_page' => 10,
    ]);
    
    foreach ( $custom_posts as $post ) {
        $sources[] = [
            'type' => 'custom_post',
            'title' => $post->post_title,
            'content' => $post->post_content,
            'url' => get_permalink( $post->ID ),
        ];
    }
    
    return $sources;
}, 10, 2 );
```
